                                                                      public class Cinema {

    private DVDPlayer dvdPlayer;

    private Screen screen;

    private Light light;

    public Cinema() {
        dvdPlayer = DVDPlayer.getInstance();
        screen = Screen.getInstance();
        light = Light.getInstance();
    }

    public void ready() {
        screen.down();
        dvdPlayer.on();
        light.dimmed();
    }

    public void play() {
        dvdPlayer.play();
    }

    public void pause() {
        dvdPlayer.pause();
    }

    public void end() {
        dvdPlayer.off();
        screen.up();
        light.brighten();
    }
}
